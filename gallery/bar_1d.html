from altair import *

Chart('http://vega.github.io/vega-lite/data/population.json').mark_bar().encode(
    x=X('sum(people):Q',
        axis=Axis(
            title='population',
        ),
    ),
).transform_data(
    filter='datum.year == 2000',
)

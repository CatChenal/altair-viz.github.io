Bubble Plot¶

< Colored Text Scatter Plot | Example Gallery | Gapminder Bubble Plot >

from altair import *

Chart('http://vega.github.io/vega-lite/data/cars.json',
    description='A bubbleplot showing horsepower on x, miles per gallons on y, and binned acceleration on size.',
).mark_point().encode(
    size='Acceleration:Q',
    x='Horsepower:Q',
    y='Miles_per_Gallon:Q',
)
